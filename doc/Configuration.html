def define_setting(name, default = nil)
  class_variable_set("@@#{name}", default)
  define_class_method "#{name}=" do |value|
    class_variable_set("@@#{name}", value)
  end
  define_class_method name do
    class_variable_get("@@#{name}")
  end
end
